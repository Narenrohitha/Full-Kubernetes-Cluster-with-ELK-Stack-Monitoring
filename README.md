### `STEP 01` — Prepare All Nodes
> 🔁 ![](https://img.shields.io/badge/Run_on-ALL_NODES-326CE5?style=flat-square&logo=kubernetes&logoColor=white)

```bash
sudo apt update && sudo apt upgrade -y

# Kubernetes requires swap OFF
sudo swapoff -a
sudo sed -i '/ swap / s/^\(.*\)$/#\1/g' /etc/fstab
```

---

### `STEP 02` — Kernel Modules & Sysctl
> 🔁 ![](https://img.shields.io/badge/Run_on-ALL_NODES-326CE5?style=flat-square&logo=kubernetes&logoColor=white)

```bash
sudo tee /etc/modules-load.d/containerd.conf <<EOF
overlay
br_netfilter
EOF
sudo modprobe overlay && sudo modprobe br_netfilter

sudo tee /etc/sysctl.d/kubernetes.conf <<EOF
net.bridge.bridge-nf-call-ip6tables = 1
net.bridge.bridge-nf-call-iptables  = 1
net.ipv4.ip_forward                 = 1
EOF
sudo sysctl --system
```

> 💡 Without `bridge-nf-call-iptables`, pod-to-pod traffic **silently fails.**

---

### `STEP 03` — Install containerd
> 🔁 ![](https://img.shields.io/badge/Run_on-ALL_NODES-326CE5?style=flat-square&logo=kubernetes&logoColor=white)

```bash
sudo apt install -y curl gnupg2 software-properties-common \
  apt-transport-https ca-certificates

sudo curl -fsSL https://download.docker.com/linux/ubuntu/gpg | \
  sudo gpg --dearmour -o /etc/apt/trusted.gpg.d/docker.gpg

sudo add-apt-repository \
  "deb [arch=amd64] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable"

sudo apt update && sudo apt install -y containerd.io

# ╔══════════════════════════════════════════════╗
# ║  CRITICAL — enable SystemdCgroup = true     ║
# ║  Skipping this → kubelet crash-loops         ║
# ╚══════════════════════════════════════════════╝
containerd config default | sudo tee /etc/containerd/config.toml >/dev/null 2>&1
sudo sed -i 's/SystemdCgroup \= false/SystemdCgroup \= true/g' \
  /etc/containerd/config.toml

sudo systemctl restart containerd && sudo systemctl enable containerd
```

> ⚠️ ![](https://img.shields.io/badge/⚠️_#1_Mistake-SystemdCgroup=false-red?style=flat-square) Set this **before** `kubeadm init`

---

### `STEP 04` — Install Kubernetes Components
> 🔁 ![](https://img.shields.io/badge/Run_on-ALL_NODES-326CE5?style=flat-square&logo=kubernetes&logoColor=white)

```bash
sudo mkdir -p /etc/apt/keyrings

echo "deb [signed-by=/etc/apt/keyrings/kubernetes-apt-keyring.gpg] \
  https://pkgs.k8s.io/core:/stable:/v1.30/deb/ /" | \
  sudo tee /etc/apt/sources.list.d/kubernetes.list

curl -fsSL https://pkgs.k8s.io/core:/stable:/v1.30/deb/Release.key | \
  sudo gpg --dearmor -o /etc/apt/keyrings/kubernetes-apt-keyring.gpg

sudo apt update
sudo apt install -y kubelet kubeadm kubectl
sudo apt-mark hold kubelet kubeadm kubectl
```

---

### `STEP 05` — Initialize Master Node
> 🎯 ![](https://img.shields.io/badge/Run_on-MASTER_ONLY-7c3aed?style=flat-square&logo=kubernetes&logoColor=white)

```bash
sudo kubeadm init

mkdir -p $HOME/.kube
sudo cp -i /etc/kubernetes/admin.conf $HOME/.kube/config
sudo chown $(id -u):$(id -g) $HOME/.kube/config
```

> 📋 **Save the `kubeadm join` command** — needed for worker nodes.

---

### `STEP 06` — Join Worker Nodes
> 🎯 ![](https://img.shields.io/badge/Run_on-EACH_WORKER-0ea5e9?style=flat-square&logo=kubernetes&logoColor=white)

```bash
kubeadm join <MASTER_IP>:6443 --token <TOKEN> \
  --discovery-token-ca-cert-hash sha256:<HASH>
```

---

### `STEP 08-10` — Build, Push & Deploy

```bash
# Build image
cat > Dockerfile <<'EOF'
FROM nginx:alpine
COPY index.html /usr/share/nginx/html/index.html
EXPOSE 80
EOF
docker build -t portfolio-site:latest .
docker tag portfolio-site:latest Narenrohitha/portfolio-site:latest
docker push Narenrohitha/portfolio-site:latest

# Deploy
kubectl apply -f deploy.yaml
# ✅ Live → http://<WORKER_IP>:30080
```

```yaml
# deploy.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: portfolio-site
  labels: { app: portfolio }
spec:
  replicas: 3
  selector:
    matchLabels: { app: portfolio }
  template:
    metadata:
      labels: { app: portfolio }
    spec:
      containers:
      - name: portfolio
        image: Narenrohitha/portfolio-site:latest
        ports: [{ containerPort: 80 }]
---
apiVersion: v1
kind: Service
metadata:
  name: portfolio-service
spec:
  type: NodePort
  selector: { app: portfolio }
  ports: [{ port: 80, targetPort: 80, nodePort: 30080 }]
```

### `STEP 13-18` — Install Full ELK Stack

```bash
helm repo add elastic https://helm.elastic.co && helm repo update
kubectl create namespace logging

# Elasticsearch
helm install elasticsearch elastic/elasticsearch \
  --namespace logging --values elasticsearch-minimal.yaml
kubectl rollout status statefulset/elasticsearch-master -n logging

# Get password
PASSWORD=$(kubectl get secret -n logging elasticsearch-master-credentials \
  -o jsonpath="{.data.password}" | base64 -d)

# Kibana
helm install kibana elastic/kibana \
  --namespace logging \
  --set service.type=NodePort \
  --set service.nodePort=30601 \
  --set elasticsearchHosts="https://elasticsearch-master:9200"

# Filebeat + Metricbeat
helm install filebeat  elastic/filebeat  --namespace logging --values filebeat-config.yaml
helm install metricbeat elastic/metricbeat --namespace logging --values metricbeat-config.yaml
```
